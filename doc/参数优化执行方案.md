# ZahnerFlow 参数优化执行方案

## 概述

本文档详细描述了ZahnerFlow系统中API配置参数的优化实施方案，采用snake_case命名约定，通过参数配置化、分组管理和自动化映射来简化参数传递流程。

## 1. 当前问题分析

### 1.1 参数传递复杂度

**当前流程**：
- 前端：15个参数需要用户配置
- 后端：15次手动参数映射
- Python：多层嵌套参数结构

**当前参数数量**：
- 核心参数：4个（必须）
- 高级参数：6个（可选）
- 专家参数：8个（很少使用）
- 总计：18个参数

### 1.2 主要问题

1. **参数转换复杂**：前端camelCase → 后端snake_case → Python snake_case
2. **用户认知负担**：18个参数都需要理解
3. **维护成本高**：每新增测量类型需要重复15次映射
4. **错误率较高**：手动映射容易出错

## 2. 优化目标

### 2.1 简化参数传递

**优化后流程**：
- 前端：统一使用snake_case命名
- 后端：直接传递，0行转换代码
- Python：平面参数结构

### 2.2 降低用户认知负担

**参数分组**：
- 基础模式：4个核心参数
- 高级模式：6个高级参数
- 专家模式：8个专家参数

### 2.3 提升维护效率

**配置化驱动**：
- 参数定义配置化
- 验证规则自动化
- UI组件动态生成

```typescript
import { PARAMETER_CONFIG } from '../config/parameters';

// 参数生成器
export class ParameterGenerator {
  // 根据配置生成节点参数定义
  static generateNodeParams(measurementType: string) {
    const config = PARAMETER_CONFIG[measurementType];
    if (!config) throw new Error(`Unknown measurement type: ${measurementType}`);

    const defaultParameters: Record<string, any> = {};
    const validationRules: Record<string, any> = {};

    // 生成默认参数
    Object.entries(config.parameters).forEach(([key, paramConfig]) => {
      defaultParameters[key] = paramConfig.default;
      validationRules[key] = {
        required: paramConfig.required,
        validation: paramConfig.validation
      };
    });

    return {
      defaultParameters,
      validationRules,
      groups: config.groups,
      dependencies: config.dependencies
    };
  }

  // 根据用户级别生成可见参数
  static getVisibleParameters(measurementType: string, userLevel: 'basic' | 'advanced' | 'expert') {
    const config = PARAMETER_CONFIG[measurementType];
    const visibleParams = new Set<string>();

    // 基础级别参数总是可见
    config.groups.basic.parameters.forEach(param => visibleParams.add(param));

    // 根据用户级别添加更多参数
    if (userLevel === 'advanced' || userLevel === 'expert') {
      config.groups.advanced.parameters.forEach(param => visibleParams.add(param));
    }

    if (userLevel === 'expert') {
      config.groups.expert.parameters.forEach(param => visibleParams.add(param));
    }

    return Array.from(visibleParams);
  }

  // 生成UI组件配置
  static generateUIConfig(measurementType: string) {
    const config = PARAMETER_CONFIG[measurementType];
    const uiConfig: Record<string, any> = {};

    Object.entries(config.parameters).forEach(([key, paramConfig]) => {
      uiConfig[key] = paramConfig.ui;
    });

    return uiConfig;
  }
}

// 参数验证器
export class ParameterValidator {
  // 验证参数
  static validate(measurementType: string, params: Record<string, any>) {
    const config = PARAMETER_CONFIG[measurementType];
    const errors: string[] = [];

    // 检查必需参数
    Object.entries(config.parameters).forEach(([key, paramConfig]) => {
      if (paramConfig.required && params[key] === undefined) {
        errors.push(`${paramConfig.ui.label} 是必需参数`);
        return;
      }

      // 如果有值，进行验证
      if (params[key] !== undefined && paramConfig.validation) {
        try {
          const isValid = paramConfig.validation(params[key]);
          if (!isValid) {
            errors.push(`${paramConfig.ui.label} 的值无效`);
          }
        } catch (error) {
          errors.push(`${paramConfig.ui.label} 验证失败: ${error.message}`);
        }
      }
    });

    // 处理参数依赖
    Object.entries(config.dependencies || {}).forEach(([param, dependency]) => {
      if (params[param] !== undefined && dependency.depends_on) {
        const dependencyValues = dependency.depends_on.map(depKey => params[depKey]);

        if (dependency.calculator) {
          const calculatedValue = dependency.calculator(...dependencyValues);
          if (Math.abs(params[param] - calculatedValue) > 0.1) {
            errors.push(`${param} 的建议值为 ${calculatedValue}`);
          }
        }
      }
    });

    return {
      isValid: errors.length === 0,
      errors
    };
  }

  // 自动计算参数建议值
  static calculateSuggestedValues(measurementType: string, providedParams: Record<string, any>) {
    const config = PARAMETER_CONFIG[measurementType];
    const suggestions: Record<string, any> = {};

    Object.entries(config.dependencies || {}).forEach(([param, dependency]) => {
      if (dependency.calculator) {
        const dependencyValues = dependency.depends_on.map(depKey => providedParams[depKey]);
        if (dependencyValues.every(value => value !== undefined)) {
          suggestions[param] = dependency.calculator(...dependencyValues);
        }
      }
    });

    return suggestions;
  }
}
```

```typescript
import React from 'react';
import { ParameterConfig } from '../config/parameters';

interface ParameterInputProps {
  paramKey: string;
  config: ParameterConfig['parameters'][string]['ui'];
  value: any;
  onChange: (value: any) => void;
  error?: string;
  suggestion?: any;
}

export const ParameterInput: React.FC<ParameterInputProps> = ({
  paramKey,
  config,
  value,
  onChange,
  error,
  suggestion
}) => {
  const handleChange = (newValue: any) => {
    onChange(newValue);
  };

  const renderInput = () => {
    switch (config.component) {
      case 'number-input':
        return (
          <div className="number-input-container">
            <input
              type="number"
              value={value}
              onChange={(e) => handleChange(parseFloat(e.target.value))}
              min={config.min}
              max={config.max}
              step={config.step}
              className={`parameter-input ${error ? 'error' : ''}`}
            />
            {config.unit && <span className="unit">{config.unit}</span>}
            {suggestion !== undefined && (
              <div className="suggestion">
                建议值: {suggestion}
                <button onClick={() => handleChange(suggestion)}>应用</button>
              </div>
            )}
          </div>
        );

      case 'path-input':
        return (
          <div className="path-input-container">
            <input
              type="text"
              value={value}
              onChange={(e) => handleChange(e.target.value)}
              placeholder={config.placeholder}
              className={`parameter-input ${error ? 'error' : ''}`}
            />
            <button onClick={() => {
              // 实现文件选择器
              const input = document.createElement('input');
              input.type = 'file';
              input.onchange = (e) => {
                const file = (e.target as HTMLInputElement).files?.[0];
                if (file) {
                  handleChange(file.path);
                }
              };
              input.click();
            }}>
              选择路径
            </button>
          </div>
        );

      case 'select-input':
        return (
          <select
            value={value}
            onChange={(e) => handleChange(e.target.value)}
            className={`parameter-input ${error ? 'error' : ''}`}
          >
            {config.options?.map((option: any) => (
              <option key={option.value} value={option.value}>
                {option.label}
              </option>
            ))}
          </select>
        );

      default:
        return (
          <input
            type="text"
            value={value}
            onChange={(e) => handleChange(e.target.value)}
            placeholder={config.placeholder}
            className={`parameter-input ${error ? 'error' : ''}`}
          />
        );
    }
  };

  return (
    <div className="parameter-input-wrapper">
      <label className="parameter-label">
        {config.label}
        {config.description && (
          <span className="parameter-description" title={config.description}>
            ?
          </span>
        )}
      </label>
      {renderInput()}
      {error && <div className="parameter-error">{error}</div>}
    </div>
  );
};
```
