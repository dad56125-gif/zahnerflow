- ✅ **零删除**: 不删除任何现有功能和代码

```typescript
// S形布局连接线计算
export const calculateAutoConnection = (
  sourceNode: ElectrochemicalNode,
  targetNode: ElectrochemicalNode,
  canvasWidth: number
): ConnectionLine => {
  const nodesPerRow = Math.max(1, Math.floor((canvasWidth - 100) / 200));
  const sourceIndex = getNodeIndex(sourceNode);
  const targetIndex = getNodeIndex(targetNode);
  const sourceRow = Math.floor(sourceIndex / nodesPerRow);
  const targetRow = Math.floor(targetIndex / nodesPerRow);

  if (sourceRow === targetRow) {
    // 同一行 - 直线连接
    return {
      type: 'straight',
      startX: sourceNode.position.x + (sourceNode.style.width || 140),
      startY: sourceNode.position.y + 30,
      endX: targetNode.position.x,
      endY: targetNode.position.y + 30
    };
  } else {
    // 不同行 - L形连接
    const isLeftToRight = sourceRow % 2 === 0;
    const startX = isLeftToRight ?
      sourceNode.position.x + (sourceNode.style.width || 140) :
      sourceNode.position.x;
    const midX = startX + (isLeftToRight ? 50 : -50);

    return {
      type: 'L-shape',
      startX,
      startY: sourceNode.position.y + 30,
      midX,
      midY: targetNode.position.y + 30,
      endX: targetNode.position.x + (targetRow % 2 === 0 ? 0 : (targetNode.style.width || 140)),
      endY: targetNode.position.y + 30
    };
  }
};
```

**循环检测算法**:
```typescript
export class LoopDetectionManager {
  static detectLoops(nodes: ElectrochemicalNode[]): LoopPair[] {
    const loopStartNodes = nodes.filter(node => node.type === 'loop_start') as LoopStartNode[];
    const loopEndNodes = nodes.filter(node => node.type === 'loop_end') as LoopEndNode[];

    const loopPairs: LoopPair[] = [];

    loopStartNodes.forEach(startNode => {
      const loopId = startNode.data.parameters.loop_id;
      const endNode = loopEndNodes.find(node => node.data.parameters.loop_id === loopId);

      if (endNode) {
        const startIndex = nodes.findIndex(node => node.id === startNode.id);
        const endIndex = nodes.findIndex(node => node.id === endNode.id);

        if (startIndex !== -1 && endIndex !== -1 && startIndex < endIndex) {
          const nodesInLoop = nodes.slice(startIndex + 1, endIndex);

          loopPairs.push({
            loopId,
            startNode,
            endNode,
            nodesInLoop,
            level: loopPairs.length
          });
        }
      }
    });

    return loopPairs;
  }
}
```

```typescript
export const NodeContextMenu: React.FC<{
  node: ElectrochemicalNode;
  position: {x: number, y: number};
  onClose: () => void;
}> = ({node, position, onClose}) => {
  const { deleteNode, duplicateNode, configureNode } = useCanvasStore();

  const handleDelete = () => {
    if (confirm(`确定要删除节点 "${node.name}" 吗？`)) {
      deleteNode(node.id);
      onClose();
    }
  };

  const handleDuplicate = () => {
    duplicateNode(node.id);
    onClose();
  };

  const handleConfigure = () => {
    configureNode(node.id);
    onClose();
  };

  return (
    <div
      className="context-menu node-context-menu"
      style={{left: position.x, top: position.y}}
    >
      <div className="context-menu-item" onClick={handleConfigure}>
        ⚙️ 配置节点
      </div>
      <div className="context-menu-item" onClick={handleDuplicate}>
        📋 复制节点
      </div>
      <div className="context-menu-separator" />
      <div className="context-menu-item danger" onClick={handleDelete}>
        🗑️ 删除节点
      </div>
    </div>
  );
};
```
