## 3. 参数同步机制

### 3.1 Python参数解析工具

**文件位置**：`apps/backend/scripts/parameter_parser.py`

**创建参数解析工具**：
```python
#!/usr/bin/env python3
"""
参数解析工具 - 从Python代码中提取API参数定义
用于同步前后端接口定义
"""

import ast
import re
import json
from typing import Dict, List, Any
from pathlib import Path

class ParameterParser:
    def __init__(self, python_file_path: str):
        self.python_file_path = python_file_path
        self.parameters = {}

    def parse_parameters(self) -> Dict[str, Any]:
        """解析Python文件中的参数定义"""
        with open(self.python_file_path, 'r', encoding='utf-8') as f:
            content = f.read()

        # 解析AST
        tree = ast.parse(content)

        # 查找所有类定义
        for node in ast.walk(tree):
            if isinstance(node, ast.ClassDef):
                # 查找Request类
                if node.name.endswith('Request'):
                    self._parse_class_parameters(node)

        return self.parameters

    def _parse_class_parameters(self, class_node: ast.ClassDef):
        """解析类参数"""
        class_name = class_node.name
        method_name = class_name.replace('Request', '').lower()

        parameters = {
            'method_name': method_name,
            'class_name': class_name,
            'parameters': {}
        }

        # 解析类属性
        for node in class_node.body:
            if isinstance(node, ast.AnnAssign):
                param_info = self._parse_parameter_annotation(node)
                if param_info:
                    parameters['parameters'][param_info['name']] = param_info

        self.parameters[method_name] = parameters

    def _parse_parameter_annotation(self, node: ast.AnnAssign) -> Dict[str, Any]:
        """解析参数注解"""
        try:
            param_name = node.target.id
            param_type = self._get_type_string(node.annotation)
            default_value = self._get_default_value(node)

            return {
                'name': param_name,
                'type': param_type,
                'default': default_value,
                'required': default_value is None
            }
        except (AttributeError, TypeError):
            return None

    def _get_type_string(self, annotation) -> str:
        """获取类型字符串"""
        if isinstance(annotation, ast.Name):
            return annotation.id
        elif isinstance(annotation, ast.Subscript):
            base = annotation.value.id
            sub = self._get_type_string(annotation.slice)
            return f"{base}[{sub}]"
        return "any"

    def _get_default_value(self, node: ast.AnnAssign) -> Any:
        """获取默认值"""
        if node.value is None:
            return None

        if isinstance(node.value, ast.Constant):
            return node.value.value
        elif isinstance(node.value, ast.Num):
            return node.value.n
        elif isinstance(node.value, ast.Str):
            return node.value.s
        elif isinstance(node.value, ast.NameConstant):
            return node.value.value
        elif isinstance(node.value, ast.List):
            return [self._get_default_value_item(item) for item in node.value.elts]

        return None

    def _get_default_value_item(self, node) -> Any:
        """获取默认值项"""
        if isinstance(node, ast.Constant):
            return node.value
        elif isinstance(node, ast.Num):
            return node.n
        elif isinstance(node, ast.Str):
            return node.s
        return None

    def generate_typescript_interface(self) -> str:
        """生成TypeScript接口定义"""
        interfaces = []

        for method_name, method_data in self.parameters.items():
            interface_name = f"{method_name.title()}Params"
            parameters = method_data['parameters']

            interface_lines = [f"export interface {interface_name} {{"]

            for param_name, param_info in parameters.items():
                type_str = self._convert_to_typescript_type(param_info['type'])
                required_str = "" if param_info['required'] else "?"
                default_str = ""

                if param_info['default'] is not None:
                    if isinstance(param_info['default'], str):
                        default_str = f" = '{param_info['default']}'"
                    else:
                        default_str = f" = {param_info['default']}"

                interface_lines.append(f"  {param_name}{required_str}: {type_str}{default_str};")

            interface_lines.append("}")
            interfaces.append("\n".join(interface_lines))

        return "\n\n".join(interfaces)

    def _convert_to_typescript_type(self, python_type: str) -> str:
        """Python类型转TypeScript类型"""
        type_mapping = {
            'str': 'string',
            'int': 'number',
            'float': 'number',
            'bool': 'boolean',
            'List': 'Array',
            'Dict': 'Record',
            'Any': 'any'
        }

        # 处理泛型类型
        if '[' in python_type and ']' in python_type:
            base_type = python_type.split('[')[0]
            sub_type = python_type.split('[')[1].split(']')[0]

            if base_type in type_mapping:
                ts_base = type_mapping[base_type]
                ts_sub = self._convert_to_typescript_type(sub_type)
                return f"{ts_base}<{ts_sub}>"

        return type_mapping.get(python_type, 'any')

    def save_to_json(self, output_path: str):
        """保存参数定义到JSON文件"""
        with open(output_path, 'w', encoding='utf-8') as f:
            json.dump(self.parameters, f, indent=2, ensure_ascii=False)

    def save_typescript_interface(self, output_path: str):
        """保存TypeScript接口到文件"""
        interface_code = self.generate_typescript_interface()
        with open(output_path, 'w', encoding='utf-8') as f:
            f.write(interface_code)

if __name__ == "__main__":
    # 使用示例
    parser = ParameterParser("zahner_device.py")
    parameters = parser.parse_parameters()

    # 保存JSON格式
    parser.save_to_json("parameters.json")

    # 生成TypeScript接口
    parser.save_typescript_interface("parameters.ts")

    print(f"解析完成，共发现 {len(parameters)} 个方法")
    print(f"参数定义已保存到 parameters.json")
    print(f"TypeScript接口已保存到 parameters.ts")
```

### 3.2 参数同步脚本

**文件位置**：`scripts/sync-parameters.py`

```python
#!/usr/bin/env python3
"""
参数同步脚本 - 自动同步Python参数定义到前端和数据库
"""

import os
import sys
import json
import shutil
from pathlib import Path
from parameter_parser import ParameterParser

def sync_parameters():
    """同步参数定义"""
    print("开始同步参数定义...")

    # 1. 解析Python文件
    python_file = "apps/backend/scripts/zahner_device.py"
    parser = ParameterParser(python_file)
    parameters = parser.parse_parameters()

    # 2. 生成TypeScript接口
    ts_interface = parser.generate_typescript_interface()

    # 3. 更新前端接口文件
    frontend_types_file = "apps/frontend/src/types/device-parameters.ts"
    update_frontend_types(frontend_types_file, ts_interface)

    # 4. 生成数据库SQL
    sql_statements = generate_database_sql(parameters)

    # 5. 保存SQL文件
    sql_file = "scripts/parameter-sync.sql"
    with open(sql_file, 'w', encoding='utf-8') as f:
        f.write(sql_statements)

    # 6. 保存参数定义JSON
    json_file = "scripts/parameters.json"
    with open(json_file, 'w', encoding='utf-8') as f:
        json.dump(parameters, f, indent=2, ensure_ascii=False)

    print(f"✅ 参数同步完成")
    print(f"📄 前端接口已更新: {frontend_types_file}")
    print(f"📊 数据库SQL已生成: {sql_file}")
    print(f"📋 参数定义已保存: {json_file}")

def update_frontend_types(file_path: str, interface_code: str):
    """更新前端TypeScript接口"""
    # 创建备份
    if os.path.exists(file_path):
        backup_path = f"{file_path}.backup"
        shutil.copy2(file_path, backup_path)
        print(f"📋 原文件已备份: {backup_path}")

    # 写入新内容
    with open(file_path, 'w', encoding='utf-8') as f:
        f.write("// 自动生成的设备参数接口定义\n")
        f.write("// 请勿手动修改，使用 scripts/sync-parameters.py 生成\n\n")
        f.write(interface_code)

def generate_database_sql(parameters: dict) -> str:
    """生成数据库同步SQL"""
    sql_statements = [
        "-- 参数同步SQL脚本",
        "-- 自动生成，请勿手动修改",
        "-- 使用 scripts/sync-parameters.py 生成",
        ""
    ]

    for method_name, method_data in parameters.items():
        sql_statements.append(f"-- 方法: {method_name}")

        # 更新参数定义表
        for param_name, param_info in method_data['parameters'].items():
            sql_statements.append(
                f"INSERT INTO parameter_definitions "
                f"(method_name, param_name, param_type, required, default_value, validation_rules) "
                f"VALUES ('{method_name}', '{param_name}', '{param_info['type']}', "
                f"{param_info['required']}, '{json.dumps(param_info['default'])}', "
                f"'{{\"type\": \"{param_info['type']}\"}}') "
                f"ON CONFLICT (method_name, param_name) DO UPDATE SET "
                f"param_type = EXCLUDED.param_type, "
                f"required = EXCLUDED.required, "
                f"default_value = EXCLUDED.default_value;"
            )

        sql_statements.append("")

    return "\n".join(sql_statements)

if __name__ == "__main__":
    sync_parameters()
```

## 4. 参数数据库实现

### 4.1 数据库表结构设计

**文件位置**：`scripts/database-init.sql`

```sql
-- 参数管理数据库初始化脚本

-- 1. 工作流表
CREATE TABLE IF NOT EXISTS workflows (
    id VARCHAR(36) PRIMARY KEY,
    name VARCHAR(255) NOT NULL,
    description TEXT,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(50) DEFAULT 'draft',
    created_by VARCHAR(100),
    version INT DEFAULT 1
);

-- 2. 节点表
CREATE TABLE IF NOT EXISTS nodes (
    id VARCHAR(36) PRIMARY KEY,
    workflow_id VARCHAR(36) NOT NULL,
    node_type VARCHAR(100) NOT NULL,
    position_x INT DEFAULT 0,
    position_y INT DEFAULT 0,
    node_config JSON,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (workflow_id) REFERENCES workflows(id) ON DELETE CASCADE
);

-- 3. 后端方法表
CREATE TABLE IF NOT EXISTS backend_methods (
    id VARCHAR(36) PRIMARY KEY,
    method_name VARCHAR(100) UNIQUE NOT NULL,
    class_name VARCHAR(100),
    description TEXT,
    api_endpoint VARCHAR(255),
    http_method VARCHAR(10) DEFAULT 'POST',
    is_active BOOLEAN DEFAULT true,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 4. 参数定义表（与Python API对齐）
CREATE TABLE IF NOT EXISTS parameter_definitions (
    id VARCHAR(36) PRIMARY KEY,
    method_name VARCHAR(100) NOT NULL,
    param_name VARCHAR(100) NOT NULL,
    param_type VARCHAR(50) NOT NULL,
    required BOOLEAN DEFAULT false,
    default_value JSON,
    validation_rules JSON,
    description TEXT,
    ui_config JSON,
    sort_order INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (method_name) REFERENCES backend_methods(method_name),
    UNIQUE(method_name, param_name)
);

-- 5. 参数组表（用户保存的参数组合）
CREATE TABLE IF NOT EXISTS parameter_groups (
    id VARCHAR(36) PRIMARY KEY,
    node_type VARCHAR(100) NOT NULL,
    method_name VARCHAR(100) NOT NULL,
    group_name VARCHAR(255) NOT NULL,
    description TEXT,
    is_default BOOLEAN DEFAULT false,
    is_public BOOLEAN DEFAULT false,
    created_by VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (method_name) REFERENCES backend_methods(method_name),
    UNIQUE(node_type, method_name, group_name, created_by)
);

-- 6. 参数组值表（参数组的默认值）
CREATE TABLE IF NOT EXISTS parameter_group_values (
    id VARCHAR(36) PRIMARY KEY,
    group_id VARCHAR(36) NOT NULL,
    param_name VARCHAR(100) NOT NULL,
    param_value JSON NOT NULL,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (group_id) REFERENCES parameter_groups(id) ON DELETE CASCADE,
    UNIQUE(group_id, param_name)
);

-- 7. 节点参数值表（实际运行时使用的参数）
CREATE TABLE IF NOT EXISTS node_parameter_values (
    id VARCHAR(36) PRIMARY KEY,
    node_id VARCHAR(36) NOT NULL,
    parameter_group_id VARCHAR(36),  -- 关联到参数组
    param_name VARCHAR(100) NOT NULL,
    param_value JSON NOT NULL,
    is_override BOOLEAN DEFAULT false,  -- 是否覆盖了参数组的默认值
    last_modified TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    modified_by VARCHAR(100),
    FOREIGN KEY (node_id) REFERENCES nodes(id) ON DELETE CASCADE,
    FOREIGN KEY (parameter_group_id) REFERENCES parameter_groups(id),
    UNIQUE(node_id, param_name)
);

-- 8. 参数变更历史表
CREATE TABLE IF NOT EXISTS parameter_change_history (
    id VARCHAR(36) PRIMARY KEY,
    node_id VARCHAR(36),
    param_name VARCHAR(100) NOT NULL,
    old_value JSON,
    new_value JSON,
    change_type VARCHAR(50),  -- 'CREATE', 'UPDATE', 'DELETE'
    changed_by VARCHAR(100),
    change_reason TEXT,
    changed_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (node_id) REFERENCES nodes(id) ON DELETE CASCADE
);

-- 9. 创建索引优化查询性能
CREATE INDEX IF NOT EXISTS idx_nodes_workflow_id ON nodes(workflow_id);
CREATE INDEX IF NOT EXISTS idx_parameter_definitions_method ON parameter_definitions(method_name);
CREATE INDEX IF NOT EXISTS idx_parameter_groups_node_type ON parameter_groups(node_type);
CREATE INDEX IF NOT EXISTS idx_node_parameter_values_node_id ON node_parameter_values(node_id);
CREATE INDEX IF NOT EXISTS idx_parameter_history_node_id ON parameter_change_history(node_id);

CREATE PROCEDURE IF NOT EXISTS sp_get_node_parameters(IN node_id VARCHAR(36))
BEGIN
    SELECT
        n.id as node_id,
        n.node_type,
        COALESCE(npv.param_value, pgv.param_value) as param_value,
        npv.is_override,
        CASE
            WHEN npv.is_override = true THEN 'custom'
            WHEN pgv.group_name IS NOT NULL THEN 'group'
            ELSE 'default'
        END as value_source
    FROM nodes n
    LEFT JOIN node_parameter_values npv ON n.id = npv.node_id
    LEFT JOIN parameter_groups pg ON npv.parameter_group_id = pg.id
    LEFT JOIN parameter_group_values pgv ON pg.id = pgv.group_id AND npv.param_name = pgv.param_name
    WHERE n.id = node_id;
END;

-- 14. 创建存储过程 - 更新节点参数
CREATE PROCEDURE IF NOT EXISTS sp_update_node_parameter(
    IN node_id VARCHAR(36),
    IN param_name VARCHAR(100),
    IN param_value JSON,
    IN modified_by VARCHAR(100)
)
BEGIN
    DECLARE old_value JSON;

    -- 获取旧值
    SELECT param_value INTO old_value
    FROM node_parameter_values
    WHERE node_id = node_id AND param_name = param_name;

    -- 更新或插入参数值
    INSERT OR REPLACE INTO node_parameter_values (id, node_id, param_name, param_value, is_override, modified_by)
    VALUES (
        COALESCE((SELECT id FROM node_parameter_values WHERE node_id = node_id AND param_name = param_name),
                 LOWER(HEX(RANDOMBLOB(4))) || '-' || LOWER(HEX(RANDOMBLOB(2))) || '-' || LOWER(HEX(RANDOMBLOB(2))) || '-' || LOWER(HEX(RANDOMBLOB(2))) || '-' || LOWER(HEX(RANDOMBLOB(6)))),
        node_id, param_name, param_value, true, modified_by
    );

    -- 记录变更历史
    IF old_value IS NOT NULL AND old_value != param_value THEN
        INSERT INTO parameter_change_history (id, node_id, param_name, old_value, new_value, change_type, changed_by)
        VALUES (
            LOWER(HEX(RANDOMBLOB(4))) || '-' || LOWER(HEX(RANDOMBLOB(2))) || '-' || LOWER(HEX(RANDOMBLOB(2))) || '-' || LOWER(HEX(RANDOMBLOB(2))) || '-' || LOWER(HEX(RANDOMBLOB(6))),
            node_id, param_name, old_value, param_value, 'UPDATE', modified_by
        );
    END IF;
END;
```

```typescript
import { Injectable, Logger } from '@nestjs/common';
import { DatabaseService } from '../database/database.service';

@Injectable()
export class ParameterService {
  private readonly logger = new Logger(ParameterService.name);

  constructor(private readonly databaseService: DatabaseService) {}

  // 获取节点完整参数
  async getNodeParameters(nodeId: string) {
    try {
      const result = await this.databaseService.query(
        `SELECT * FROM sp_get_node_parameters(?)`,
        [nodeId]
      );

      // 将结果转换为参数对象
      const parameters: Record<string, any> = {};
      result.forEach(row => {
        parameters[row.param_name] = {
          value: JSON.parse(row.param_value),
          source: row.value_source,
          isOverride: row.is_override
        };
      });

      return parameters;
    } catch (error) {
      this.logger.error(`获取节点参数失败: ${error.message}`);
      throw error;
    }
  }

  // 更新节点参数
  async updateNodeParameter(
    nodeId: string,
    paramName: string,
    paramValue: any,
    modifiedBy: string = 'system'
  ) {
    try {
      await this.databaseService.query(
        `CALL sp_update_node_parameter(?, ?, ?, ?)`,
        [nodeId, paramName, JSON.stringify(paramValue), modifiedBy]
      );

      this.logger.log(`节点参数更新成功: ${nodeId}.${paramName} = ${paramValue}`);

      return { success: true, message: '参数更新成功' };
    } catch (error) {
      this.logger.error(`更新节点参数失败: ${error.message}`);
      throw error;
    }
  }

  // 获取可用的参数组
  async getParameterGroups(nodeType: string, method_name: string) {
    try {
      const result = await this.databaseService.query(
        `SELECT id, group_name, description, is_default, created_by
         FROM parameter_groups
         WHERE node_type = ? AND method_name = ?
         ORDER BY is_default DESC, created_at DESC`,
        [nodeType, method_name]
      );

      return result;
    } catch (error) {
      this.logger.error(`获取参数组失败: ${error.message}`);
      throw error;
    }
  }

  // 创建参数组
  async createParameterGroup(groupData: {
    nodeType: string;
    methodName: string;
    groupName: string;
    description?: string;
    isDefault?: boolean;
    parameters: Record<string, any>;
    createdBy: string;
  }) {
    try {
      const groupId = this.generateUUID();

      // 创建参数组
      await this.databaseService.query(
        `INSERT INTO parameter_groups (id, node_type, method_name, group_name, description, is_default, created_by)
         VALUES (?, ?, ?, ?, ?, ?, ?)`,
        [
          groupId,
          groupData.nodeType,
          groupData.methodName,
          groupData.groupName,
          groupData.description || '',
          groupData.isDefault || false,
          groupData.createdBy
        ]
      );

      // 插入参数组值
      for (const [paramName, paramValue] of Object.entries(groupData.parameters)) {
        await this.databaseService.query(
          `INSERT INTO parameter_group_values (id, group_id, param_name, param_value)
           VALUES (?, ?, ?, ?)`,
          [this.generateUUID(), groupId, paramName, JSON.stringify(paramValue)]
        );
      }

      this.logger.log(`参数组创建成功: ${groupData.groupName}`);

      return { success: true, groupId, message: '参数组创建成功' };
    } catch (error) {
      this.logger.error(`创建参数组失败: ${error.message}`);
      throw error;
    }
  }

  // 设置节点参数组
  async setNodeParameterGroup(nodeId: string, groupId: string, modifiedBy: string = 'system') {
    try {
      // 获取参数组的所有参数
      const groupParameters = await this.databaseService.query(
        `SELECT param_name, param_value FROM parameter_group_values WHERE group_id = ?`,
        [groupId]
      );

      // 批量设置节点参数
      for (const param of groupParameters) {
        await this.updateNodeParameter(
          nodeId,
          param.param_name,
          JSON.parse(param.param_value),
          modifiedBy
        );
      }

      // 更新节点的parameter_group_id
      await this.databaseService.query(
        `UPDATE node_parameter_values
         SET parameter_group_id = ?
         WHERE node_id = ?`,
        [groupId, nodeId]
      );

      this.logger.log(`节点参数组设置成功: ${nodeId} -> ${groupId}`);

      return { success: true, message: '参数组设置成功' };
    } catch (error) {
      this.logger.error(`设置节点参数组失败: ${error.message}`);
      throw error;
    }
  }

  // 获取参数变更历史
  async getParameterHistory(nodeId: string, paramName?: string) {
    try {
      let query = `
        SELECT * FROM parameter_change_history
        WHERE node_id = ?
      `;
      const params = [nodeId];

      if (paramName) {
        query += ` AND param_name = ?`;
        params.push(paramName);
      }

      query += ` ORDER BY changed_at DESC`;

      const result = await this.databaseService.query(query, params);

      return result.map(row => ({
        ...row,
        oldValue: JSON.parse(row.old_value || '{}'),
        newValue: JSON.parse(row.new_value || '{}')
      }));
    } catch (error) {
      this.logger.error(`获取参数历史失败: ${error.message}`);
      throw error;
    }
  }

  private generateUUID(): string {
    return 'xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx'.replace(/[xy]/g, function(c) {
      const r = Math.random() * 16 | 0;
      const v = c == 'x' ? r : (r & 0x3 | 0x8);
      return v.toString(16);
    });
  }
}
```
